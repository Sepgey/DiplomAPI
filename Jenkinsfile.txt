properties ([disableConcurrentBuilds()])

pipeline
{
	agent
	{
		label 'docker'
	}
	options
	{
		buildDiscarder(logRotator(numToKeepStr: '10', artifactNumToKeepStr: '10'))
		timestamps()
	}
	stages
	{
		stage("create docker image")
		{
			steps
			{
				echo "======= start building image ======="
				dir ('DiplomAPI/Docker')
				{
					sh 'docker build . '
				}
			}
		}
	}
}
